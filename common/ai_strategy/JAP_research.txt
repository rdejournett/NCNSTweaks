JAP_change_ship_priority = {

	allowed = {
		has_dlc = "Man the Guns"
		original_tag = JAP
	}
	enable = {
		date < 1939.1.1
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = dockyard_to_military_factory_ratio
		value = 20  #aim for a 2-1 ratio on naval factories
	}

		type = role_ratio
		id = naval_capital
		value = -100
	}

	ai_strategy = {
		type = role_ratio
		id = naval_carrier
		value = -50
	}

	ai_strategy = {
		type = role_ratio
		id = naval_sub
		value = 20
	}

	ai_strategy = {
		type = role_ratio
		id = naval_screen
		value = 100
	}

}

# improve DD research tech



JAP_improve_DD_upgrade = {
    allowed = { tag = JAP }
    enable = { always = yes }
    abort = { always = no }

# Forces the AI to use Naval XP to create new blueprints
	ai_strategy = {
		type = navy_xp_spend_priority
		id = equipment_variant
		value = 1000
	}

	# Tells the AI to prioritize "upgrading" existing lines to the newest tech
	ai_strategy = {
		type = equipment_variant_production_factor
		id = destroyer
		value = 200
	}
}
