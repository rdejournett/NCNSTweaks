

ALL_reduce_land_priority_guns_2 = {
    allowed = {
		original_tag = ITA
	}
    
    # Correct trigger for equipment amount
    enable = {
        has_equipment = {
            infantry_equipment > 20000
        }
    }
    
    # Added to stop the "missing abort" error
    abort = {
        NOT = {
            has_equipment = {
                infantry_equipment > 1000
            }
        }
    }

	
    ai_strategy = {
        type = equipment_production_factor
        id = infantry
        value = -1
    }
	ai_strategy = {
		type = role_ratio
		id = infantry
		value = 0
	}
}

##### globals


# Universal Strategy to kill land overproduction once stockpiles are full
GLOBAL_stop_infantry = {
    allowed = { is_ai = yes }
    enable = {
        has_equipment = { infantry_equipment > 50000 }
    }
    abort_when_not_enabled = yes
    ai_strategy = { type = equipment_production_factor id = infantry value = 0.01 }
    ai_strategy = { type = role_ratio id = infantry value = -50 }
}

GLOBAL_stop_support = {
    allowed = { is_ai = yes }
    enable = {
        has_equipment = { support_equipment > 5000 }
    }
    abort_when_not_enabled = yes
    ai_strategy = { type = equipment_production_factor id = support value = 0.01 }
}


GLOBAL_stop_tanks = {
    allowed = { is_ai = yes }
    enable = {
        OR = {
            has_equipment = { light_tank_chassis > 1000 }
			has_equipment = { medium_tank_chassis > 1000 }
			has_equipment = { heavy_tank_chassis > 1000 }
			has_equipment = { modern_tank_chassis > 1000 }
        }
    }
    abort_when_not_enabled = yes
    ai_strategy = { type = equipment_production_factor id = armor value = 0.05 }
    ai_strategy = { type = role_ratio id = armor value = -100 }
}


GLOBAL_stop_air = {
    allowed = { is_ai = yes }
    enable = {
		OR = {
            has_equipment = { small_plane_airframe > 1000 }
			has_equipment = { medium_plane_airframe > 1000 }
        }
    }
    abort_when_not_enabled = yes
    ai_strategy = { type = equipment_production_factor id = fighter value = 0.1 }
    ai_strategy = { type = equipment_production_factor id = interceptor value = 0.1 }
	ai_strategy = { type = equipment_production_factor id = cas value = 0.1 }
	ai_strategy = { type = equipment_production_factor id = naval_bomber value = 0.1 }
	
}
