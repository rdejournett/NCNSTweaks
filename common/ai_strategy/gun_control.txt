



##### globals


# Universal Strategy to kill land overproduction once stockpiles are full
GLOBAL_stop_infantry = {
    allowed = { is_ai = yes }
    enable = {
		date > 1945.1.1
        has_equipment = { infantry_equipment > 50000 }
    }
    abort_when_not_enabled = yes
    ai_strategy = { type = equipment_production_factor id = infantry value = 0.01 }
    ai_strategy = { type = role_ratio id = infantry value = -50 }
}

GLOBAL_stop_support = {
    allowed = { is_ai = yes }
    enable = {
		date > 1945.1.1
        has_equipment = { support_equipment > 5000 }
    }
    abort_when_not_enabled = yes
    ai_strategy = { type = equipment_production_factor id = support value = 0.01 }
}


GLOBAL_stop_tanks = {
    allowed = { is_ai = yes }
    enable = {
		date > 1945.1.1
        OR = {
            has_equipment = { light_tank_chassis > 1000 }
			has_equipment = { medium_tank_chassis > 1000 }
			has_equipment = { heavy_tank_chassis > 1000 }
			has_equipment = { modern_tank_chassis > 1000 }
        }
    }
    abort_when_not_enabled = yes
    ai_strategy = { type = equipment_production_factor id = armor value = 0.05 }
    ai_strategy = { type = role_ratio id = armor value = -100 }
}
