
on_actions = {
    on_startup = {
        effect = {
            # Move from Global scope to Italy's scope
            ITA = {
				navy_experience = 250
            }
            JAP = {
				navy_experience = 250
            }			
        }
		effect = {
            ITA = {
				if = {
					limit = {
						tag = ITA
						NOT = { has_idea = ITA_super_dockyards }
					}
					add_ideas = ITA_super_dockyards
				}
				
				add_dynamic_modifier = {
					modifier = ITA_navy_modifier
					scope = ITA
				}

			}
            JAP = {
				if = {
					limit = {
						tag = JAP
						NOT = { has_idea = JAP_super_dockyards }
					}
					add_ideas = JAP_super_dockyards
				}
				
				add_dynamic_modifier = {
					modifier = JAP_navy_modifier
					scope = JAP
				}

			}
        }
    }

	
    # Fires every month at 00:00 (Safety check)
    on_monthly = {
        effect = {
            if = {
                limit = {
                    tag = ITA
                    NOT = { has_idea = ITA_super_dockyards }
                }
                add_ideas = ITA_super_dockyards
            }
        }
    }


	
}
